### Database Schema

```sql
-- Advisors table
CREATE TABLE advisors (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    uuid CHAR(36) UNIQUE NOT NULL,
    name VARCHAR(255) NOT NULL,
    title VARCHAR(255) NOT NULL,
    expertise_area VARCHAR(255) NOT NULL,
    description TEXT NOT NULL,
    persona_prompt TEXT NOT NULL,
    system_instructions TEXT NOT NULL,
    
    -- Generated characteristics
    personality_traits JSON NOT NULL,
    communication_style JSON NOT NULL,
    knowledge_domains JSON NOT NULL,
    response_patterns JSON NOT NULL,
    
    -- Quality metrics
    authenticity_score INT DEFAULT NULL,
    validation_attempts INT DEFAULT 0,
    last_validated_at TIMESTAMP NULL,
    
    -- Performance tracking
    conversation_count INT DEFAULT 0,
    average_rating DECIMAL(3,2) DEFAULT NULL,
    total_interactions INT DEFAULT 0,
    
    -- Metadata
    is_active BOOLEAN DEFAULT TRUE,
    is_public BOOLEAN DEFAULT TRUE,
    version INT DEFAULT 1,
    created_by BIGINT NULL,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (created_by) REFERENCES users(id) ON DELETE SET NULL,
    
    INDEX idx_expertise_area (expertise_area),
    INDEX idx_is_active_public (is_active, is_public),
    INDEX idx_authenticity_score (authenticity_score),
    INDEX idx_created_by (created_by)
);

-- Conversations table
CREATE TABLE conversations (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    uuid CHAR(36) UNIQUE NOT NULL,
    user_id BIGINT NOT NULL,
    advisor_id BIGINT NOT NULL,
    
    -- Session management
    title VARCHAR(255) DEFAULT NULL,
    status ENUM('active', 'completed', 'archived') DEFAULT 'active',
    context_summary TEXT DEFAULT NULL,
    
    -- Message tracking
    message_count INT DEFAULT 0,
    total_tokens_used INT DEFAULT 0,
    
    -- Performance metrics
    user_rating INT DEFAULT NULL, -- 1-5 scale
    completion_reason ENUM('user_ended', 'timeout', 'error', 'natural') DEFAULT NULL,
    
    -- Timestamps
    started_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_message_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ended_at TIMESTAMP NULL,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (advisor_id) REFERENCES advisors(id) ON DELETE CASCADE,
    
    INDEX idx_user_id (user_id),
    INDEX idx_advisor_id (advisor_id),
    INDEX idx_status (status),
    INDEX idx_started_at (started_at),
    INDEX idx_user_advisor (user_id, advisor_id)
);

-- Conversation messages table
CREATE TABLE conversation_messages (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    conversation_id BIGINT NOT NULL,
    role ENUM('user', 'advisor') NOT NULL,
    content TEXT NOT NULL,
    
    -- Message metadata
    token_count INT DEFAULT NULL,
    processing_time_ms INT DEFAULT NULL,
    model_used VARCHAR(100) DEFAULT NULL,
    
    -- Context tracking
    context_used TEXT DEFAULT NULL,
    previous_context_id BIGINT DEFAULT NULL,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (conversation_id) REFERENCES conversations(id) ON DELETE CASCADE,
    FOREIGN KEY (previous_context_id) REFERENCES conversation_messages(id) ON DELETE SET NULL,
    
    INDEX idx_conversation_id (conversation_id),
    INDEX idx_role (role),
    INDEX idx_created_at (created_at)
);

-- User-Advisor relationships (favorites, custom advisors, etc.)
CREATE TABLE user_advisors (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    advisor_id BIGINT NOT NULL,
    relationship_type ENUM('favorite', 'custom', 'blocked') NOT NULL,
    
    -- Custom preferences
    custom_instructions TEXT DEFAULT NULL,
    notification_preferences JSON DEFAULT NULL,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    UNIQUE KEY unique_user_advisor_type (user_id, advisor_id, relationship_type),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (advisor_id) REFERENCES advisors(id) ON DELETE CASCADE,
    
    INDEX idx_user_id (user_id),
    INDEX idx_advisor_id (advisor_id),
    INDEX idx_relationship_type (relationship_type)
);

-- Advisor validation logs (from Task 004)
CREATE TABLE advisor_validation_logs (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    advisor_id BIGINT NOT NULL,
    validation_type VARCHAR(50) NOT NULL,
    score INT NOT NULL,
    passed BOOLEAN NOT NULL,
    details JSON DEFAULT NULL,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (advisor_id) REFERENCES advisors(id) ON DELETE CASCADE,
    
    INDEX idx_advisor_id (advisor_id),
    INDEX idx_validation_type (validation_type),
    INDEX idx_passed (passed),
    INDEX idx_created_at (created_at)
);
```

### Eloquent Models

```php
// app/Models/Advisor.php
class Advisor extends Model
{
    protected $fillable = [
        'uuid', 'name', 'title', 'expertise_area', 'description',
        'persona_prompt', 'system_instructions', 'personality_traits',
        'communication_style', 'knowledge_domains', 'response_patterns',
        'is_active', 'is_public', 'created_by'
    ];

    protected $casts = [
        'personality_traits' => 'array',
        'communication_style' => 'array', 
        'knowledge_domains' => 'array',
        'response_patterns' => 'array',
        'is_active' => 'boolean',
        'is_public' => 'boolean',
        'last_validated_at' => 'datetime',
        'average_rating' => 'decimal:2'
    ];

    // Relationships
    public function conversations(): HasMany
    public function creator(): BelongsTo
    public function favoritedByUsers(): BelongsToMany
    public function validationLogs(): HasMany

    // Scopes
    public function scopeActive(Builder $query): Builder
    public function scopePublic(Builder $query): Builder
    public function scopeByExpertise(Builder $query, string $area): Builder
}

// app/Models/Conversation.php
class Conversation extends Model
{
    protected $fillable = [
        'uuid', 'user_id', 'advisor_id', 'title', 'status',
        'context_summary', 'user_rating', 'completion_reason'
    ];

    protected $casts = [
        'user_rating' => 'integer',
        'started_at' => 'datetime',
        'last_message_at' => 'datetime', 
        'ended_at' => 'datetime'
    ];

    // Relationships
    public function user(): BelongsTo
    public function advisor(): BelongsTo
    public function messages(): HasMany

    // Scopes
    public function scopeActive(Builder $query): Builder
    public function scopeRecent(Builder $query): Builder
}

// app/Models/ConversationMessage.php
class ConversationMessage extends Model
{
    protected $fillable = [
        'conversation_id', 'role', 'content', 'token_count',
        'processing_time_ms', 'model_used', 'context_used'
    ];

    protected $casts = [
        'token_count' => 'integer',
        'processing_time_ms' => 'integer'
    ];

    // Relationships
    public function conversation(): BelongsTo
    public function previousContext(): BelongsTo
}

// app/Models/UserAdvisor.php
class UserAdvisor extends Model
{
    protected $fillable = [
        'user_id', 'advisor_id', 'relationship_type',
        'custom_instructions', 'notification_preferences'
    ];

    protected $casts = [
        'notification_preferences' => 'array'
    ];

    // Relationships
    public function user(): BelongsTo
    public function advisor(): BelongsTo
}
```
