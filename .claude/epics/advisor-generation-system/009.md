<?php

namespace App\Services\Conversation;

use App\Models\Conversation;
use App\Models\Message;
use App\Models\ConversationContext;
use App\Models\ConversationMemory;
use Illuminate\Support\Collection;
use Illuminate\Support\Facades\DB;

class ConversationManager
{
    public function __construct(
        private ContextManager $contextManager,
        private MemoryManager $memoryManager,
        private HistoryManager $historyManager,
        private AnalyticsManager $analyticsManager
    ) {}

    public function createConversation(array $data): Conversation
    {
        $conversation = DB::transaction(function () use ($data) {
            $conversation = Conversation::create([
                'user_id' => $data['user_id'],
                'title' => $data['title'],
                'description' => $data['description'] ?? null,
                'settings' => $data['settings'] ?? null,
                'memory_key' => $this->generateMemoryKey($data['user_id']),
                'importance_score' => $this->calculateInitialImportance($data),
            ]);

            // Initialize conversation context
            $this->contextManager->initializeContext($conversation);

            // Load relevant memories
            $this->memoryManager->loadRelevantMemories($conversation);

            return $conversation;
        });

        $this->analyticsManager->recordConversationCreated($conversation);

        return $conversation;
    }

    public function switchAdvisor(
        Conversation $conversation, 
        int $newAdvisorId, 
        ?string $handoffReason = null
    ): array {
        return DB::transaction(function () use ($conversation, $newAdvisorId, $handoffReason) {
            // Create context snapshot
            $contextSnapshot = $this->contextManager->createSnapshot(
                $conversation,
                'advisor_handoff',
                [
                    'previous_advisor_id' => $conversation->current_advisor_id,
                    'new_advisor_id' => $newAdvisorId,
                    'handoff_reason' => $handoffReason,
                ]
            );

            // Prepare context for new advisor
            $handoffContext = $this->contextManager->prepareHandoffContext(
                $conversation,
                $newAdvisorId,
                $contextSnapshot
            );

            // Update conversation
            $conversation->update([
                'current_advisor_id' => $newAdvisorId,
                'context_summary' => $handoffContext['summary'],
                'updated_at' => now(),
            ]);

            return [
                'success' => true,
                'context' => $handoffContext,
                'snapshot_id' => $contextSnapshot->id,
            ];
        });
    }

    public function addMessage(Conversation $conversation, array $messageData): Message
    {
        return DB::transaction(function () use ($conversation, $messageData) {
            // Prepare context window
            $contextWindow = $this->contextManager->getContextWindow(
                $conversation,
                $messageData['role'] === 'user' ? 'incoming' : 'outgoing'
            );

            // Create message
            $message = $conversation->messages()->create([
                'role' => $messageData['role'],
                'content' => $messageData['content'],
                'advisor_id' => $messageData['advisor_id'] ?? null,
                'council_session_id' => $messageData['council_session_id'] ?? null,
                'context_window' => $contextWindow,
                'importance_score' => $this->calculateMessageImportance($messageData, $conversation),
                'thread_id' => $messageData['thread_id'] ?? null,
            ]);

            // Update conversation statistics
            $this->updateConversationStats($conversation, $message);

            // Process for memory extraction
            if ($message->role === 'assistant') {
                $this->memoryManager->extractAndStoreMemories($message, $conversation);
            }

            // Update context summary if needed
            if ($this->shouldUpdateContext($conversation)) {
                $this->contextManager->updateContextSummary($conversation);
            }

            return $message;
        });
    }

    public function searchConversations(int $userId, array $filters): Collection
    {
        $query = Conversation::where('user_id', $userId)
            ->where('status', '!=', 'deleted');

        // Apply filters
        if (!empty($filters['search'])) {
            $query->whereRaw('MATCH(title, description, context_summary) AGAINST(? IN NATURAL LANGUAGE MODE)', 
                [$filters['search']]);
        }

        if (!empty($filters['category'])) {
            $query->where('category', $filters['category']);
        }

        if (!empty($filters['tags'])) {
            foreach ($filters['tags'] as $tag) {
                $query->whereJsonContains('tags', $tag);
            }
        }

        if (!empty($filters['date_range'])) {
            $query->whereBetween('created_at', [
                $filters['date_range']['start'],
                $filters['date_range']['end']
            ]);
        }

        if (!empty($filters['advisor_id'])) {
            $query->whereHas('messages', function ($q) use ($filters) {
                $q->where('advisor_id', $filters['advisor_id']);
            });
        }

        if (isset($filters['is_favorite'])) {
            $query->where('is_favorite', $filters['is_favorite']);
        }

        // Sort by relevance or date
        $sortBy = $filters['sort_by'] ?? 'updated_at';
        $sortDirection = $filters['sort_direction'] ?? 'desc';

        if ($sortBy === 'relevance' && !empty($filters['search'])) {
            $query->selectRaw('*, MATCH(title, description, context_summary) AGAINST(? IN NATURAL LANGUAGE MODE) as relevance_score', 
                [$filters['search']])
                ->orderBy('relevance_score', 'desc');
        } else {
            $query->orderBy($sortBy, $sortDirection);
        }

        return $query->with(['messages' => function ($q) {
            $q->latest()->take(3); // Load recent messages for preview
        }, 'advisors'])->paginate($filters['per_page'] ?? 20);
    }

    public function exportConversation(Conversation $conversation, string $format = 'json'): array
    {
        $this->authorize('view', $conversation);

        $exportData = [
            'conversation' => $conversation->only([
                'uuid', 'title', 'description', 'created_at', 'updated_at'
            ]),
            'messages' => $conversation->messages()
                ->with('advisor:id,name')
                ->orderBy('created_at')
                ->get()
                ->map(function ($message) {
                    return [
                        'role' => $message->role,
                        'content' => $message->content,
                        'advisor' => $message->advisor?->name,
                        'created_at' => $message->created_at,
                    ];
                }),
            'statistics' => [
                'total_messages' => $conversation->message_count,
                'total_tokens' => $conversation->total_tokens_used,
                'duration' => $conversation->created_at->diffForHumans($conversation->last_message_at),
            ],
        ];

        return match ($format) {
            'pdf' => $this->exportToPdf($exportData),
            'markdown' => $this->exportToMarkdown($exportData),
            'txt' => $this->exportToPlainText($exportData),
            default => $exportData,
        };
    }

    private function generateMemoryKey(int $userId): string
    {
        return "user_{$userId}_" . now()->format('Y_m');
    }

    private function calculateInitialImportance(array $data): float
    {
        // Base importance score calculation
        $score = 0.5; // Default neutral importance

        // Adjust based on conversation type
        if (isset($data['settings']['mode'])) {
            $score += match ($data['settings']['mode']) {
                'council' => 0.2, // Councils are typically more important
                'research' => 0.1,
                default => 0.0,
            };
        }

        // Adjust based on explicit priority
        if (isset($data['priority'])) {
            $score += match ($data['priority']) {
                'high' => 0.2,
                'medium' => 0.1,
                'low' => -0.1,
                default => 0.0,
            };
        }

        return min(1.0, max(0.0, $score));
    }

    private function calculateMessageImportance(array $messageData, Conversation $conversation): float
    {
        $score = 0.5; // Base score

        // Length-based importance
        $contentLength = strlen($messageData['content']);
        if ($contentLength > 1000) $score += 0.1;
        if ($contentLength > 2000) $score += 0.1;

        // Role-based importance
        if ($messageData['role'] === 'system') $score += 0.2;

        // Council session importance
        if (!empty($messageData['council_session_id'])) $score += 0.2;

        return min(1.0, max(0.0, $score));
    }

    private function shouldUpdateContext(Conversation $conversation): bool
    {
        // Update context every 10 messages or if significant time has passed
        return ($conversation->message_count % 10 === 0) ||
               ($conversation->updated_at->diffInHours(now()) > 24);
    }

    private function updateConversationStats(Conversation $conversation, Message $message): void
    {
        $conversation->increment('message_count');
        $conversation->increment('total_tokens_used', $message->tokens_used ?? 0);
        $conversation->update(['last_message_at' => $message->created_at]);
    }
}

    public function getContextWindow(Conversation $conversation, string $direction = 'both'): array
    {
        $baseQuery = $conversation->messages()
            ->with('advisor:id,name')
            ->orderBy('created_at', 'desc');

        // Get recent messages based on conversation settings
        $windowSize = $conversation->settings['context_window_size'] ?? 10;
        $recentMessages = $baseQuery->take($windowSize)->get();

        // Include important messages from further back
        $importantMessages = $conversation->messages()
            ->where('importance_score', '>', 0.8)
            ->whereNotIn('id', $recentMessages->pluck('id'))
            ->orderBy('created_at', 'desc')
            ->take(5)
            ->get();

        $allMessages = $recentMessages->merge($importantMessages)
            ->sortBy('created_at')
            ->values();

        return [
            'messages' => $allMessages->map(function ($message) {
                return [
                    'id' => $message->id,
                    'role' => $message->role,
                    'content' => $this->truncateContent($message->content, 500),
                    'advisor' => $message->advisor?->name,
                    'created_at' => $message->created_at,
                    'importance' => $message->importance_score,
                ];
            })->toArray(),
            'total_tokens' => $allMessages->sum('tokens_used'),
            'summary' => $conversation->context_summary,
        ];
    }

            ->orderBy('created_at', 'desc')
            ->take(20)
            ->get();

        // Generate new context summary using AI
        $summary = $this->generateContextSummary($recentMessages->toArray());
        
        $conversation->update(['context_summary' => $summary]);
    }

    private function compressContext(Conversation $conversation): array
    {
        $messages = $conversation->messages()
            ->orderBy('created_at', 'desc')
            ->take(50)
            ->get();

        $originalTokens = $messages->sum('tokens_used');
        
        // Compress by extracting key information
        $compressed = [
            'key_topics' => $this->extractKeyTopics($messages),
            'important_decisions' => $this->extractImportantDecisions($messages),
            'user_preferences' => $this->extractUserPreferences($messages),
            'context_flow' => $this->extractConversationFlow($messages),
            'recent_context' => $messages->take(5)->map(function ($message) {
                return [
                    'role' => $message->role,
                    'summary' => $this->summarizeContent($message->content),
                    'importance' => $message->importance_score,
                ];
            })->toArray(),
        ];

        $compressedTokens = $this->estimateTokenCount($compressed);

        return [
            'data' => $compressed,
            'original_tokens' => $originalTokens,
            'compressed_tokens' => $compressedTokens,
            'compression_ratio' => $originalTokens > 0 ? $compressedTokens / $originalTokens : 1,
        ];
    }

    private function tailorContextForAdvisor(array $contextData, $advisor): array
    {
        $expertise = $advisor->expertise_areas ?? [];
        
        // Filter and prioritize context based on advisor expertise

            $contextData['data']['key_topics'] ?? [],
            fn($topic) => $this->isRelevantToExpertise($topic, $expertise)
        );

        return [
            'summary' => $this->generateTailoredSummary($contextData['data'], $expertise),
            'key_points' => $relevantTopics,
            'advisor_specific' => [
                'relevant_decisions' => $this->filterRelevantDecisions(
                    $contextData['data']['important_decisions'] ?? [],
                    $expertise
                ),
                'context_focus' => $this->determineContextFocus($expertise, $contextData['data']),
            ],
        ];
    }

    private function truncateContent(string $content, int $maxLength): string
    {
        return strlen($content) > $maxLength

            : $content;
    }
