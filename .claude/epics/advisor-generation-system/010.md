// app/Services/Cache/LayeredCacheManager.php
class LayeredCacheManager
{
    public function __construct(
        private CacheManager $cacheManager,
        private MetricsCollector $metrics
    ) {}
    
    public function remember(string $key, int $ttl, callable $callback, array $layers = ['l1', 'l2']): mixed
    {
        $startTime = microtime(true);
        
        foreach ($layers as $layer) {
            $store = $this->cacheManager->store($this->getStoreForLayer($layer));
            
            if ($store->has($key)) {
                $this->metrics->increment('cache.hit', ['layer' => $layer]);
                $this->recordCacheMetrics($key, $layer, microtime(true) - $startTime, true);
                return $store->get($key);
            }
        }
        
        // Cache miss - execute callback and store in all layers
        $this->metrics->increment('cache.miss');
        $value = $callback();
        
        foreach (array_reverse($layers) as $layer) {
            $store = $this->cacheManager->store($this->getStoreForLayer($layer));
            $layerTtl = $this->getTtlForLayer($layer, $ttl);
            $store->put($key, $value, $layerTtl);
        }
        
        $this->recordCacheMetrics($key, 'miss', microtime(true) - $startTime, false);
        return $value;
    }
    
    public function invalidate(string $pattern): void
    {
        // Invalidate across all cache layers
        foreach (['l1', 'l2', 'l3'] as $layer) {
            $store = $this->cacheManager->store($this->getStoreForLayer($layer));
            
            if (method_exists($store, 'deleteByPattern')) {
                $store->deleteByPattern($pattern);
            }
        }
        
        $this->metrics->increment('cache.invalidation', ['pattern' => $pattern]);
    }
}

class GenerateAdvisorResponse implements ShouldQueue, ShouldBeUnique
{
    use Queueable, InteractsWithQueue, SerializesModels;
    
    public int $tries = 3;
    public int $maxExceptions = 2;
    public int $timeout = 120;
    public array $backoff = [10, 30, 60]; // Exponential backoff in seconds
    
    public function __construct(
        private Message $message,
        private Advisor $advisor,
        private array $context = []
    ) {
        // Set queue based on priority
        $this->onQueue(config('queue.job_priorities.advisor_response', 'default'));
        
        // Unique job key to prevent duplicates
        $this->uniqueId = "advisor_response_{$this->message->id}";
    }
    
    public function handle(AdvisorService $advisorService, PerformanceMonitor $monitor): void
    {
        $monitor->startOperation('advisor_response_generation', [
            'message_id' => $this->message->id,
            'advisor_id' => $this->advisor->id,
        ]);
        
        try {
            // Generate response with streaming
            $responseStream = $advisorService->generateStreamingResponse(
                $this->message,
                $this->advisor,
                $this->context
            );
            
            // Process stream chunks
            foreach ($responseStream as $chunk) {
                // Broadcast chunk to WebSocket clients
                broadcast(new ResponseChunkGenerated($this->message, $chunk));
                
                // Update message with accumulated response
                $this->message->update([
                    'content' => $this->message->content . $chunk,
                    'status' => 'streaming',
                ]);
            }
            
            // Finalize response
            $this->message->update([
                'status' => 'completed',
                'completed_at' => now(),
            ]);
            
            broadcast(new ResponseCompleted($this->message));
            
            $monitor->endOperation('advisor_response_generation', ['status' => 'success']);
            
        } catch (\Exception $e) {
            $this->message->update([
                'status' => 'failed',
                'error_message' => $e->getMessage(),
            ]);
            
            $monitor->endOperation('advisor_response_generation', [
                'status' => 'error',
                'error' => $e->getMessage()
            ]);
            
            broadcast(new ResponseFailed($this->message, $e->getMessage()));
            
            throw $e; // Re-throw for queue retry logic
        }
    }
    
    public function failed(\Throwable $exception): void
    {
        $this->message->update([
            'status' => 'failed',
            'error_message' => $exception->getMessage(),
        ]);
        
        broadcast(new ResponseFailed($this->message, $exception->getMessage()));
        
        // Log for monitoring
        Log::error('Advisor response generation failed', [
            'message_id' => $this->message->id,
            'advisor_id' => $this->advisor->id,
            'exception' => $exception->getMessage(),
            'attempts' => $this->attempts(),
        ]);
    }
}

// app/Services/Streaming/ResponseStreamer.php
class ResponseStreamer
{
    public function __construct(
        private RedisManager $redis,
        private BroadcastManager $broadcaster
    ) {}
    
    public function streamResponse(
        Message $message,
        \Generator $responseGenerator,
        array $options = []
    ): void {
        $channel = "conversation.{$message->conversation_id}";
        $chunkSize = $options['chunk_size'] ?? 50;
        $buffer = '';
        $totalTokens = 0;
        
        foreach ($responseGenerator as $chunk) {
            $buffer .= $chunk;
            $totalTokens++;
            
            // Send chunk when buffer reaches optimal size
            if (strlen($buffer) >= $chunkSize) {
                $this->broadcastChunk($channel, [
                    'message_id' => $message->id,
                    'chunk' => $buffer,
                    'tokens' => $totalTokens,
                    'complete' => false,
                ]);
                
                // Store chunk for recovery
                $this->storeChunkForRecovery($message->id, $buffer, $totalTokens);
                
                $buffer = '';
            }
        }
        
        // Send final chunk
        if (!empty($buffer)) {
            $this->broadcastChunk($channel, [
                'message_id' => $message->id,
                'chunk' => $buffer,
                'tokens' => $totalTokens,
                'complete' => false,
            ]);
            
            $this->storeChunkForRecovery($message->id, $buffer, $totalTokens);
        }
        
        // Send completion signal
        $this->broadcastChunk($channel, [
            'message_id' => $message->id,
            'complete' => true,
            'total_tokens' => $totalTokens,
        ]);
        
        // Cleanup recovery data
        $this->cleanupRecoveryData($message->id);
    }
    
    public function recoverStreamingResponse(int $messageId): ?array
    {
        $recoveryKey = "streaming_recovery:{$messageId}";
        $chunks = $this->redis->lrange($recoveryKey, 0, -1);
        
        if (empty($chunks)) {
            return null;
        }
        
        $fullContent = '';
        $totalTokens = 0;
        
        foreach ($chunks as $chunkData) {
            $chunk = json_decode($chunkData, true);
            $fullContent .= $chunk['content'];
            $totalTokens = $chunk['tokens'];
        }
        
        return [
            'content' => $fullContent,
            'tokens' => $totalTokens,
            'recovered' => true,
        ];
    }
    
    private function broadcastChunk(string $channel, array $data): void
    {
        $this->broadcaster->channel($channel)->send($data);
        
        // Also send via Server-Sent Events for fallback
        $this->sendSSE($channel, $data);
    }
    
    private function storeChunkForRecovery(int $messageId, string $chunk, int $tokens): void
    {
        $recoveryKey = "streaming_recovery:{$messageId}";
        $chunkData = json_encode([
            'content' => $chunk,
            'tokens' => $tokens,
            'timestamp' => time(),
        ]);
        
        $this->redis->lpush($recoveryKey, $chunkData);
        $this->redis->expire($recoveryKey, 3600); // Keep for 1 hour
    }
    
    private function sendSSE(string $channel, array $data): void
    {
        $sseData = "data: " . json_encode($data) . "\n\n";
        
        // Store SSE data in Redis for clients to retrieve
        $sseKey = "sse:{$channel}";
        $this->redis->lpush($sseKey, $sseData);
        $this->redis->expire($sseKey, 300); // Keep for 5 minutes
        $this->redis->ltrim($sseKey, 0, 99); // Keep last 100 messages
    }
}

class QueryOptimizer
{
    public function __construct(
        private CacheManager $cache,
        private MetricsCollector $metrics
    ) {}
    
    public function optimizeConversationQueries(): void
    {
        // Add composite indexes for common query patterns
        DB::statement('
            CREATE INDEX IF NOT EXISTS idx_conversations_user_status_updated 
            ON conversations (user_id, status, updated_at DESC)
        ');
        
        DB::statement('
            CREATE INDEX IF NOT EXISTS idx_messages_conversation_created 
            ON messages (conversation_id, created_at DESC)
        ');
        
        DB::statement('
            CREATE INDEX IF NOT EXISTS idx_messages_importance_created 
            ON messages (importance_score DESC, created_at DESC)
        ');
    }
    
    public function cacheableQuery(string $key, callable $query, int $ttl = 300): mixed
    {
        return $this->cache->remember($key, $ttl, function () use ($query) {
            $startTime = microtime(true);
            $result = $query();
            $duration = (microtime(true) - $startTime) * 1000;
            
            $this->metrics->timing('database.query_time', $duration);
            $this->metrics->increment('database.query_count');
            
            return $result;
        });
    }
    
    public function batchInsert(string $table, array $data, int $batchSize = 1000): void
    {
        $chunks = array_chunk($data, $batchSize);
        
        foreach ($chunks as $chunk) {
            DB::table($table)->insert($chunk);
        }
        
        $this->metrics->increment('database.batch_insert', [
            'table' => $table,
            'total_records' => count($data),
            'batches' => count($chunks),
        ]);
    }
}

// app/Services/Monitoring/PerformanceMonitor.php
class PerformanceMonitor
{
    private array $activeOperations = [];
    
    public function __construct(
        private MetricsCollector $metrics,
        private AlertManager $alerts
    ) {}
    
    public function startOperation(string $operation, array $context = []): string
    {
        $operationId = uniqid($operation . '_', true);
        
        $this->activeOperations[$operationId] = [
            'operation' => $operation,
            'context' => $context,
            'start_time' => microtime(true),
            'start_memory' => memory_get_usage(),
        ];
        
        return $operationId;
    }
    
    public function endOperation(string $operationId, array $additionalData = []): void
    {
        if (!isset($this->activeOperations[$operationId])) {
            return;
        }
        
        $operation = $this->activeOperations[$operationId];
        $duration = (microtime(true) - $operation['start_time']) * 1000; // ms
        $memoryUsed = memory_get_usage() - $operation['start_memory'];
        
        // Record metrics
        $this->metrics->timing("performance.{$operation['operation']}.duration", $duration);
        $this->metrics->gauge("performance.{$operation['operation']}.memory", $memoryUsed);
        
        // Check for SLA violations
        $this->checkSLAViolation($operation['operation'], $duration);
        
        // Log performance data
        Log::channel('performance')->info('Operation completed', [
            'operation' => $operation['operation'],
            'duration_ms' => $duration,
            'memory_bytes' => $memoryUsed,
            'context' => $operation['context'],
            'additional_data' => $additionalData,
        ]);
        
        unset($this->activeOperations[$operationId]);
    }
    
    public function trackUserExperience(string $userId, string $action, float $duration): void
    {
        $this->metrics->timing("user_experience.{$action}", $duration, [
            'user_id' => $userId,
        ]);
        
        // Store in time-series for trend analysis
        $this->storeUserExperienceMetric($userId, $action, $duration);
    }
    
    public function detectPerformanceRegression(): array
    {
        $currentMetrics = $this->getLastHourMetrics();
        $baselineMetrics = $this->getBaselineMetrics();
        
        $regressions = [];
        
        foreach ($currentMetrics as $metric => $value) {
            if (isset($baselineMetrics[$metric])) {
                $baseline = $baselineMetrics[$metric];
                $percentageIncrease = (($value - $baseline) / $baseline) * 100;
                
                if ($percentageIncrease > 20) { // 20% regression threshold
                    $regressions[] = [
                        'metric' => $metric,
                        'current' => $value,
                        'baseline' => $baseline,
                        'regression' => $percentageIncrease,
                    ];
                }
            }
        }
        
        if (!empty($regressions)) {
            $this->alerts->send('performance_regression', [
                'regressions' => $regressions,
                'timestamp' => now(),
            ]);
        }
        
        return $regressions;
    }
    
    private function checkSLAViolation(string $operation, float $duration): void
    {
        $slaThresholds = [
            'advisor_response_generation' => 30000, // 30 seconds
            'council_orchestration' => 45000, // 45 seconds
            'conversation_load' => 1000, // 1 second
            'search_query' => 500, // 500ms
        ];
        
        if (isset($slaThresholds[$operation]) && $duration > $slaThresholds[$operation]) {
            $this->alerts->send('sla_violation', [
                'operation' => $operation,
                'duration' => $duration,
                'threshold' => $slaThresholds[$operation],
                'violation_percentage' => (($duration - $slaThresholds[$operation]) / $slaThresholds[$operation]) * 100,
            ]);
        }
    }
}
