```php
// Broadcasting events
class AdvisorResponseGenerated implements ShouldBroadcast
{
    public function broadcastOn(): array
    {
        return [
            new PrivateChannel("conversation.{$this->message->conversation_id}"),
        ];
    }
    
    public function broadcastWith(): array
    {
        return [
            'message_id' => $this->message->id,
            'content' => $this->message->content,
            'advisor' => $this->message->advisor->only(['id', 'name']),
            'created_at' => $this->message->created_at,
        ];
    }
}

// Server-sent events for response streaming
Route::get('/stream/conversations/{conversation}/messages/{message}', function (Conversation $conversation, Message $message) {
    return response()->stream(function () use ($message) {
        while ($message->status === 'processing') {
            $chunk = $message->getNextChunk(); // Get next response chunk
            echo "data: " . json_encode(['chunk' => $chunk]) . "\n\n";
            flush();
            sleep(1);
        }
        echo "data: " . json_encode(['complete' => true]) . "\n\n";
    }, 200, [
        'Content-Type' => 'text/plain',
        'Cache-Control' => 'no-cache',
    ]);
})->middleware('auth:sanctum');
```

### Error Handling
```php
// Standardized API error responses
class ApiErrorHandler
{
    public function render($request, Throwable $exception)
    {
        if ($request->expectsJson()) {
            return match (true) {
                $exception instanceof ModelNotFoundException => response()->json([
                    'error' => 'Resource not found',
                    'message' => 'The requested resource could not be found.',
                    'code' => 'RESOURCE_NOT_FOUND'
                ], 404),
                
                $exception instanceof ValidationException => response()->json([
                    'error' => 'Validation failed',
                    'message' => 'The given data was invalid.',
                    'errors' => $exception->errors(),
                    'code' => 'VALIDATION_ERROR'
                ], 422),
                
                $exception instanceof AdvisorUnavailableException => response()->json([
                    'error' => 'Advisor unavailable',
                    'message' => 'The requested advisor is currently unavailable.',
                    'code' => 'ADVISOR_UNAVAILABLE',
                    'retry_after' => 60
                ], 503),
                
                default => response()->json([
                    'error' => 'Server error',
                    'message' => 'An unexpected error occurred.',
                    'code' => 'INTERNAL_ERROR'
                ], 500)
            };
        }
        
        return parent::render($request, $exception);
    }
}
```

### Controller Design Patterns
```php
class AdvisorController extends Controller
{
    public function index(Request $request): AdvisorCollection
    {
        $advisors = Advisor::query()
            ->when($request->expertise, fn($q, $expertise) => 
                $q->whereJsonContains('expertise_areas', $expertise))
            ->when($request->search, fn($q, $search) => 
                $q->where('name', 'like', "%{$search}%"))
            ->active()
            ->paginate($request->per_page ?? 20);
            
        return new AdvisorCollection($advisors);
    }
    
    public function store(StoreAdvisorRequest $request): AdvisorResource
    {
        $advisor = Advisor::create($request->validated());
        
        // Test advisor configuration
        if (!$this->advisorService->testConfiguration($advisor)) {
            throw new AdvisorConfigurationException('Invalid advisor configuration');
        }
        
        return new AdvisorResource($advisor);
    }
}
```

```php
class AdvisorResource extends JsonResource
{
    public function toArray($request): array
    {
        return [
            'id' => $this->id,
            'uuid' => $this->uuid,
            'name' => $this->name,
            'description' => $this->description,
            'expertise_areas' => $this->expertise_areas,
            'personality_traits' => $this->personality_traits,
            'is_active' => $this->is_active,
            'created_at' => $this->created_at,
            
            // Conditional fields
            'system_prompt' => $this->when(
                $request->user()?->isAdmin(), 
                $this->system_prompt
            ),
            
            'model_config' => $this->when(
                $request->user()?->isAdmin(), 
                $this->model_config

                    'average_rating' => $this->average_rating,
                    'total_tokens_used' => $this->total_tokens_used,

        ];
    }
```
